The let should be properly aligned in the following code:

\begin{code}%
\>[0]\AgdaFunction{id} \AgdaSymbol{:} \AgdaSymbol{(}\AgdaBound{A} \AgdaSymbol{:} \AgdaPrimitiveType{Set}\AgdaSymbol{)} \AgdaSymbol{→} \AgdaBound{A} \AgdaSymbol{→} \AgdaBound{A}\<%
\\
\>[0]\AgdaFunction{id} \AgdaBound{A} \AgdaSymbol{=}\<%
\\
\>[0][@{}l@{\AgdaIndent{0}}]\<[2]%
\>[2]\AgdaKeyword{let}\<%
\\
\>[2][@{}l@{\AgdaIndent{0}}]\<[4]%
\>[4]\AgdaBound{id'} \AgdaSymbol{:} \AgdaFunction{B} \AgdaSymbol{→} \AgdaFunction{B}\<%
\\
\>[2][@{}l@{\AgdaIndent{0}}]\<[4]%
\>[4]\AgdaBound{id'} \AgdaSymbol{=} \AgdaSymbol{λ} \AgdaBound{a} \AgdaSymbol{→} \AgdaBound{a}\<%
\\
\>[0][@{}l@{\AgdaIndent{0}}]\<[2]%
\>[2]\AgdaKeyword{in} \AgdaBound{id'}\<%
\\
\>[0][@{}l@{\AgdaIndent{0}}]\<[2]%
\>[2]\AgdaKeyword{where}\<%
\\
\>[2][@{}l@{\AgdaIndent{0}}]\<[4]%
\>[4]\AgdaFunction{B} \AgdaSymbol{:} \AgdaPrimitiveType{Set}\<%
\\
\>[2][@{}l@{\AgdaIndent{0}}]\<[4]%
\>[4]\AgdaFunction{B} \AgdaSymbol{=} \AgdaBound{A}\<%
\end{code}
